#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{
 int temp = 0;
 int number[6];
 
   for (int i = 0; i < 6; i++) {
     srand(time(NULL));
     number[i] = (int)(rand() % 45) + 1;
     for(int j = 0; j < i; j++) {
       if (number[i] == number[j]) {
         i--;
       }
     }
     printf("%d ", number[i]);
     sleep(1);
   }
 printf("\n**********************\n");
   for (int i = 0 ; i < 6; i++) {  
     for (int j = 0 ; j < 5 ; j++) {
       if (number[j] > number[j+1]) {
         temp = number[j];
         number[j] = number[j+1];
         number[j+1] = temp;
       }
     }
   }

   for (int i = 0 ; i < 6; i++) {
     printf("%d ", number[i]);  
   }
    return 0;
}
