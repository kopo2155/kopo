#include <stdio.h>  
int main()
{
   int a = 0;
   int b = 0; 
   int c = 0;
   int result = 0;
   
   scanf("%d", & a);
   scanf("%d", & b);
   scanf("%d", & c);
   if (c == 0){
     printf("세번째 값이 잘못되었습니다.");
   }
   else{
     result = (a + b) * c - (a - b) / c ;
     printf("연산결과 : (%d + %d) * %d - (%d - %d) / %d = %d",
             a, b, c, a, b, c, result);
   }
    return 0;
}
