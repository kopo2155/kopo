   int a = 0;
   int b = 0;
   int c = 0;
   int result = 0;
   
   scanf("%d", & a);
   scanf("%d", & b);
   scanf("%d", & c);
   result = ( a + b) * c - (a - b) / c ;
   if (c < 0){
      printf("0이상을 입력하세요.");}
   else{
      printf("연산결과 : ( %d + %d) * %d - (%d - %d) / %d = %d",
             a, b, c, a, b, c, result);
   }
