#include <stdio.h>

int main()
{
   float a = 0;
   int A = 0;
   scanf("%f", &a);
   if( (int)a >= 30) {
     A = 1;
   } else if( (int)a >= 25) {
     A = 2;
   } else if( (int)a >= 15) {
     A = 3;
   } else if( (int)a >= 5) {
     A = 4;
   } else if( (int)a >= 0) {
     A = 5;
   } else if( (int)a < 0) {
     A = 6;
   }   
                 
   switch ((int)A)
   {
     case(1):
       printf("%0.2f도 : 너무 더워요! 반팔과 반바지를 입으세요.", a);
       break;
     case(2):
       printf("%0.2f도 : 따뜻한 날씨입니다! 가벼운 옷을 입으세요.", a);
       break;
     case(3):
       printf("%0.2f도 : 선선한 날씨입니다! 가벼운 겉옷을 챙기세요.", a);
       break;
     case(4):
       printf("%0.2f도 : 조금 쌀쌀합니다! 따뜻한 옷을 입으세요.", a);
       break;
     case(5):
       printf("%0.2f도 : 매우 춥습니다! 두꺼운 외투를 꼭 입으세요.", a);
       break;
     case(6):
       printf("%0.2f도 : 한파 경고! 패딩과 장갑, 모자를 착용하세요.", a);
       break;
   }
    return 0;
}
