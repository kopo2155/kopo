i = float(input('행을 입력하시오.'))
j = float(input('열을 입력하시요.'))
n = 0
if i == int(i) and j == int(j) and int(i) > 0 and int(j) > 0:
  for i in range(1, int(i + 1)):
    for k in range(0,int(j)):
      if n + k != n + int(j) - 1:
        print(n + 1 + k, end = "")
      else :
        print(n + 1 + k)
    n = n + 1 + k 
else:
  print("잘못된 수입니다.")
